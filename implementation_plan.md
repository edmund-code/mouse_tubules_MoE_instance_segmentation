    boundary_width: int = 2,
    include_inner_boundaries: bool = True
) -> np.ndarray:
    """
    Generate boundary map from instance mask.
    
    Parameters
    ----------
    instance_mask : np.ndarray
        Instance mask of shape (H, W).
    boundary_width : int, default=2
        Width of boundary in pixels.
    include_inner_boundaries : bool, default=True
        If True, include boundaries between touching instances.
        If False, only include outer boundaries of foreground.
    
    Returns
    -------
    np.ndarray
        Binary boundary mask of shape (H, W) with dtype np.int64.
        Non-boundary = 0, boundary = 1.

            - Add to combined boundary map
        - This creates boundaries at the edge of EACH instance, including where they touch
    
    - If include_inner_boundaries is False:
        - Convert to semantic mask first
        - Compute single morphological gradient on semantic mask
        - Misses boundaries between touching instances
    
    - Clip final values to [0, 1]
    - Use skimage.morphology.disk() for structuring element
    - Use skimage.morphology.dilation() and erosion()
    """

def instance_to_sdf(
    instance_mask: np.ndarray,
    normalize: bool = True,
    truncate: float = None
) -> np.ndarray:
    """
    Generate signed distance field from instance mask.
    
    The SDF is computed such that:
    - Pixels inside instances have NEGATIVE values (distance to nearest boundary)
    - Pixels outside instances have POSITIVE values (distance to nearest boundary)
    - Boundary pixels have value close to 0
    
    Parameters
    ----------
    instance_mask : np.ndarray
        Instance mask of shape (H, W).
    normalize : bool, default=True
        If True, normalize SDF to range [-1, 1] using tanh-like normalization.
    truncate : float, optional
        If provided, truncate absolute SDF values to this maximum.
        Applied before normalization.
    
    Returns
    -------
    np.ndarray
        SDF of shape (H, W) with dtype np.float32.
    
    Implementation Notes
    --------------------
    - Method 1 (Instance-aware, RECOMMENDED):
        - For each unique instance ID (excluding 0):
            - Create binary mask for this instance
            - Compute distance transform for inside: scipy.ndimage.distance_transform_edt(mask)
            - Compute distance transform for outside: scipy.ndimage.distance_transform_edt(~mask)
            - SDF for this instance = outside_dist - inside_dist
        - For each pixel, use the SDF value from its own instance
        - For background pixels, use minimum distance to any instance (positive)
    
    - Method 2 (Simpler, semantic-based):
        - Convert to semantic mask
        - inside_dist = distance_transform_edt(semantic_mask)
        - outside_dist = distance_transform_edt(~semantic_mask)
        - SDF = outside_dist - inside_dist
        - This doesn't distinguish between instances
    
    - Use Method 1 for better instance separation
    
    - If truncate is provided: sdf = np.clip(sdf, -truncate, truncate)
    
    - If normalize is True:
        - Option A: sdf = sdf / (np.abs(sdf).max() + 1e-8)
        - Option B: sdf = np.tanh(sdf / scale) where scale is a hyperparameter
        - Use Option A for simplicity
    """

def variance_loss(
    embeddings: torch.Tensor,
    instance_mask: torch.Tensor,
    instance_means: torch.Tensor,
    num_instances: int,
    delta_v: float = 0.5
) -> torch.Tensor:
    """
    Compute variance loss that pulls embeddings toward their instance mean.
    
    L_var = (1/C) * Σ_c [ (1/N_c) * Σ_i [ max(0, ||μ_c - x_i|| - δ_v)² ] ]
    
    Where:
    - C is number of instances
    - N_c is number of pixels in instance c
    - μ_c is mean embedding of instance c
    - x_i is embedding of pixel i belonging to instance c
    - δ_v is the variance margin (distance within which loss is 0)
    
    Parameters
    ----------
    embeddings : torch.Tensor
        Embedding tensor of shape (E, H, W).
    instance_mask : torch.Tensor
        Instance mask of shape (H, W).
    instance_means : torch.Tensor
        Instance means of shape (num_instances, E).
    num_instances : int
        Number of instances.
    delta_v : float, default=0.5
        Variance margin. Embeddings within this distance of mean incur no loss.
    
    Returns
    -------
    torch.Tensor
        Scalar variance loss.
    
    Implementation Notes
    --------------------
    - For each instance c (1 to num_instances):
        - Get pixels belonging to this instance
        - Get embeddings for these pixels: shape (N_c, E)
        - Get mean for this instance: shape (E,)
        - Compute L2 distance from each pixel embedding to mean
        - Apply hinge: max(0, distance - delta_v)
        - Square the result
        - Average over pixels in this instance
    - Average over all instances
    - Use torch.clamp(x, min=0) for max(0, x)
    - Use torch.norm(x, dim=-1) for L2 norm
    """

def distance_loss(
    instance_means: torch.Tensor,
    num_instances: int,
    delta_d: float = 1.5
) -> torch.Tensor:
    """
    Compute distance loss that pushes instance means apart.
    
    L_dist = (1 / C(C-1)) * Σ_cA Σ_cB [ max(0, 2*δ_d - ||μ_cA - μ_cB||)² ]
    
    Where:
    - C is number of instances
    - μ_cA, μ_cB are means of instances cA and cB
    - δ_d is the distance margin
    
    Parameters
    ----------
    instance_means : torch.Tensor
        Instance means of shape (num_instances, E).
    num_instances : int
        Number of instances.
    delta_d : float, default=1.5
        Distance margin. Means farther than 2*delta_d incur no loss.
    
    Returns
    -------
    torch.Tensor
        Scalar distance loss.
    
    Implementation Notes
    --------------------
    - If num_instances <= 1, return 0 (no pairs to push apart)
    - Compute pairwise distances between all instance means
        - Use: dists = torch.cdist(means, means)  # shape (C, C)
    - Apply hinge: max(0, 2*delta_d - distance)
    - Square the result
    - Sum upper triangular (excluding diagonal) to avoid double counting
    - Normalize by C*(C-1)/2 (number of pairs)
    - Alternatively, use nested loops for clarity (slower but clearer)
    """

    Returns
    -------
    tuple

    - For each pred-gt pair, compute:
      intersection = np.sum((pred_mask == pred_id) & (gt_mask == gt_id))
      union = np.sum((pred_mask == pred_id) | (gt_mask == gt_id))
      iou = intersection / union if union > 0 else 0
    - This can be slow for many instances; optimize with vectorization if needed

    """
    Compute Aggregated Jaccard Index (AJI).
    
    AJI = Σ |G_i ∩ P_σ(i)| / Σ |G_i ∪ P_σ(i)| + Σ |P_k|
    
    Where:
    - G_i are ground truth instances
    - P_σ(i) is the predicted instance best matching G_i
    - P_k are unmatched predicted instances
    
    Parameters
    ----------
    pred_mask : np.ndarray
        Predicted instance mask of shape (H, W).
    gt_mask : np.ndarray
        Ground truth instance mask of shape (H, W).
    
    Returns
    -------
    float
        AJI score in range [0, 1]. Higher is better.
    
    Implementation Notes
    --------------------
    - For each GT instance, find pred instance with highest IoU (greedy matching)
    - Accumulate intersection and union for matched pairs
    - Add pixels from unmatched predictions to denominator
    - Handle edge cases: no GT, no pred, empty masks
    """

def compute_panoptic_quality(
    pred_mask: np.ndarray,
    gt_mask: np.ndarray,
    iou_threshold: float = 0.5
) -> dict[str, float]:
    """
    Compute Panoptic Quality (PQ) and its components.
    
    PQ = (Σ IoU(p, g)) / (|TP| + 0.5*|FP| + 0.5*|FN|)
       = SQ * RQ
    
    Where:
    - SQ (Segmentation Quality) = mean IoU of matched pairs
    - RQ (Recognition Quality) = TP / (TP + 0.5*FP + 0.5*FN)
    
    Parameters
    ----------
    pred_mask : np.ndarray
        Predicted instance mask.
    gt_mask : np.ndarray
        Ground truth instance mask.
    iou_threshold : float
        IoU threshold for matching.
    
    Returns
    -------
    dict[str, float]
        {
            "pq": Panoptic Quality,
            "sq": Segmentation Quality,
            "rq": Recognition Quality,
            "tp": Number of true positives,
            "fp": Number of false positives,
            "fn": Number of false negatives
        }
    
    Implementation Notes
    --------------------
    - Use compute_iou_matrix() and match_instances()
    - TP = number of valid matches
    - FP = number of unmatched predictions
    - FN = number of unmatched ground truth
    - SQ = mean IoU of TP matches (0 if no TP)
    - RQ = TP / (TP + 0.5*FP + 0.5*FN)
    - PQ = SQ * RQ
    """

    annotation_handler: 'QuPathAnnotationHandler',
    patch_size: int = 256,
    stride: int = 256,
    min_tissue_ratio: float = 0.1,

    """
    Extract patches from a WSI with corresponding instance masks.
    
    Parameters
    ----------
    wsi_image : np.ndarray
        Full WSI image of shape (H, W, 3).
    annotation_handler : QuPathAnnotationHandler
        Handler for QuPath annotations.
    patch_size : int
        Size of patches to extract.
    stride : int
        Stride between patches. Use stride < patch_size for overlap.
    min_tissue_ratio : float
        Minimum ratio of patch that must be tissue (non-white).
        Filters out background patches.
    min_annotation_ratio : float
        Minimum ratio of patch that must have annotations.
        Set to 0 to include unannotated patches (for unsupervised).
    
    Returns
    -------
    list[dict]
        List of patch dictionaries:
        {
            "image": np.ndarray (patch_size, patch_size, 3),
            "instance_mask": np.ndarray (patch_size, patch_size),
            "x": int (top-left x coordinate in WSI),
            "y": int (top-left y coordinate in WSI),
            "num_instances": int
        }
    
    Implementation Notes
    --------------------
    1. Calculate grid of patch positions based on WSI size, patch_size, stride
    2. For each position (x, y):
       a. Extract image patch: wsi_image[y:y+patch_size, x:x+patch_size]
       b. Check tissue ratio:
          - Convert to grayscale
          - Threshold to find tissue (non-white pixels)
          - If tissue_ratio < min_tissue_ratio, skip patch
       c. Get instance mask: annotation_handler.create_mask_for_region(
             bounds=(x, y, x+patch_size, y+patch_size),
             mask_shape=(patch_size, patch_size)
          )
       d. Check annotation ratio:
          - annotation_ratio = np.sum(instance_mask > 0) / (patch_size ** 2)
          - If annotation_ratio < min_annotation_ratio, skip (or include for unsup)
       e. Add to results list
    3. Return list of valid patches
    """

def train_one_epoch(
    epoch: int,
    models: dict,
    optimizers: dict,
    dataloaders: dict,
    loss_functions: dict,
    args,
    cfg,
    writer: SummaryWriter
) -> dict:
    """
    Train for one epoch.
    
    Parameters
    ----------
    epoch : int
        Current epoch number.
    models : dict
        All model components.
    optimizers : dict
        All optimizers.
    dataloaders : dict
        All data loaders.
    loss_functions : dict
        All loss functions.
    args : argparse.Namespace
        Arguments.
    cfg : dict
        Configuration.
    writer : SummaryWriter
        TensorBoard writer.
    
    Returns
    -------
    dict
        Training metrics for this epoch.
    
    Implementation Notes (follow original train.py structure):
    --------------------
    1. Set all models to train mode
    2. Initialize loss accumulators
    3. Compute unsup_weight for this epoch (curriculum)
    4. Iterate through batches:
    
       For each batch:
       a. UNSUPERVISED FORWARD PASS:
          - Get unsupervised batch
          - Forward through all 4 experts
          - Concatenate features, pass through gating
          - Generate pseudo-labels from gating outputs:
            * fake_mask = argmax(seg_out)
            * fake_sdf = tanh(sdf_out)
            * fake_bnd = argmax(bnd_out)
            * fake_instances = generate from boundary + semantic (connected components)
          - Compute unsupervised losses for all 4 tasks
          - Scale by unsup_weight
          - Backward (retain_graph=True)
          
       b. SUPERVISED FORWARD PASS:
          - Get supervised batch (includes instance_mask, semantic_mask, SDF, boundary)
          - Forward through all 4 experts
          - Concatenate features, pass through gating
          - Compute supervised losses:
            * seg_loss: on both expert output and gating output
            * sdf_loss: on both expert output and gating output
            * bnd_loss: on both expert output and gating output
            * embed_loss: on both expert output and gating output
          - Combine with MultiTaskLoss
          - Backward
          
       c. OPTIMIZER STEP:
          - Step all optimizers
          - Clear gradients
          - Empty CUDA cache
          
       d. ACCUMULATE LOSSES for logging
    
    5. Step all schedulers
    6. Log to TensorBoard:
       - Scalars: all loss components
       - Images: sample predictions (every N epochs)
    7. Return metrics dict
    """

Common Issues and Solutions
Issue 1: CUDA Out of Memory
Reduce batch_size from 4 to 2
Reduce embedding_dim from 16 to 8
Add gradient checkpointing to experts
Issue 2: GeoJSON Coordinate Mismatch
QuPath uses (x, y) coordinates, numpy uses (row, col) = (y, x)
Always check coordinate order when converting
Issue 3: Empty Instance Masks
Check min_overlap_ratio in get_annotations_in_region()
Verify GeoJSON class names match filter
Issue 4: Embedding Clustering Too Slow
Reduce embedding_dim
Downsample embeddings before clustering
Use HDBSCAN instead of mean-shift
Issue 5: Instance IDs Not Consecutive
Always call renumber_instances() after filtering
Performance Optimization Tips
Pre-compute labels: Generate SDF/boundary during patch extraction, not during training
Use mixed precision: Add torch.cuda.amp for faster training
Increase num_workers: Use 8-16 workers for data loading
Cache embeddings: During validation, cache embeddings to disk for analysis
